  5def cluster_distribution(
  6    adata: AnnData,
  7    group: str,
  8    cluster: str,
  9    normalize: bool = True
 10) -> pd.DataFrame:
 11    """
 12    Calculate the distribution of clusters across groups in AnnData object.
 13
 14    This function computes either the counts or percentages of cells from each cluster
 15    that are present in each group. The result is a matrix where rows represent clusters
 16    and columns represent groups.
 17
 18    Parameters
 19    ----------
 20    adata : AnnData
 21        Annotated data matrix containing cell annotations.
 22    group : str
 23        Column name in adata.obs containing group labels (e.g., 'condition', 'treatment').
 24    cluster : str
 25        Column name in adata.obs containing cluster labels (e.g., 'leiden', 'louvain').
 26    normalize : bool, optional (default=True)
 27        If True, return percentages (0-100).
 28        If False, return absolute counts.
 29
 30    Returns
 31    -------
 32    pd.DataFrame
 33        Matrix of cluster distributions across groups.
 34        Rows: clusters
 35        Columns: groups
 36        Values: percentages (if normalize=True) or counts (if normalize=False)
 37
 38    Raises
 39    ------
 40    ValueError
 41        If group or cluster columns are not found in adata.obs
 42    RuntimeError
 43        If an error occurs during calculation
 44
 45    Examples
 46    --------
 47    >>> import scanpy as sc
 48    >>> adata = sc.datasets.pbmc3k()
 49    >>> sc.tl.leiden(adata)
 50    >>> result = cluster_distribution(adata, group='bulk_labels', cluster='leiden')
 51    >>> print(result)
 52    """
 53    # Validate input columns exist in adata.obs
 54    if group not in adata.obs.columns:
 55        raise ValueError(f"Group column '{group}' not found in adata.obs")
 56    if cluster not in adata.obs.columns:
 57        raise ValueError(f"Cluster column '{cluster}' not found in adata.obs")
 58
 59    try:
 60        # Calculate distribution using pandas operations
 61        grouped = adata.obs[[group, cluster]].groupby(group, observed=True)
 62        value_counts = grouped.value_counts(normalize=normalize)
 63        
 64        # Convert to percentages if normalized
 65        if normalize:
 66            value_counts = value_counts * 100
 67            rounded_counts = value_counts.round(1)
 68        else:
 69            rounded_counts = value_counts
 70
 71        # Reshape result to matrix form and fill missing values
 72        result = rounded_counts.unstack().fillna(0).transpose()
 73        return result
 74    
 75    except Exception as e:
 76        raise RuntimeError(f"Error calculating cluster distribution: {str(e)}")

 78def ensemble_identify_cluster_markers(
 79    adata: AnnData,
 80    methods: list = ["wilcoxon", "t-test_overestim_var", "t-test", "logreg"],
 81    var_cluster: str = "cluster",
 82    n_cores: int = -1
 83) -> pd.DataFrame:
 84    """
 85    Perform ensemble differential expression analysis using multiple statistical methods.
 86
 87    This function combines results from multiple differential expression methods to identify
 88    robust marker genes for each cluster. It calculates a consensus ranking based on the
 89    median ranks across all methods used.
 90
 91    Parameters
 92    ----------
 93    adata : AnnData
 94        Annotated data matrix that must contain:
 95        - 'highly_variable' column in .var identifying variable genes
 96        - var_cluster column in .obs containing cluster labels
 97    methods : list, optional
 98        Statistical methods to use for differential expression analysis.
 99        Must be valid methods for scanpy.tl.rank_genes_groups.
100        Default: ["wilcoxon", "t-test_overestim_var", "t-test", "logreg"]
101    n_cores : int, optional
102        Number of CPU cores for parallel processing.
103        Default: -1 (use all available cores)
104
105    Returns
106    -------
107    pd.DataFrame
108        Combined results with the following columns:
109        - names: Gene names
110        - cluster: Cluster identifiers
111        - {method}_rank: Rankings from each method
112        - {method}_scores: Scores from each method
113        - {method}_pvals: P-values (if available)
114        - {method}_pvals_adj: Adjusted p-values (if available)
115        - logfoldchanges: Mean log fold changes across methods
116        - pct_group: Mean percentage of cells expressing gene in cluster
117        - pct_rest: Mean percentage of cells expressing gene in other clusters
118        - pct_diff: Difference between pct_group and pct_rest
119        - median_rank: Consensus ranking across all methods
120
121    Notes
122    -----
123    - Analysis is performed only on highly variable genes
124    - Results are filtered by adjusted p-value < 0.05 when available
125    - Final rankings are 1-based within each cluster
126    - Results are sorted by cluster and median rank
127
128    Examples
129    --------
130    >>> import scanpy as sc
131    >>> adata = sc.datasets.pbmc3k()
132    >>> sc.pp.highly_variable_genes(adata)
133    >>> sc.tl.leiden(adata, key_added=var_cluster)
134    >>> markers = ensemble_identify_cluster_markers(adata)
135    >>> print(markers.head())
136    """
137    # Make a copy and subset to HVG
138    # Check if 'highly_variable' exists, if not, compute it
139    if "highly_variable" not in adata.var.columns:
140        print("Column 'highly_variable' not found in adata.var. Running sc.pp.highly_variable_genes...")
141        sc.pp.highly_variable_genes(adata)
142    adata_copy = adata[:, adata.var["highly_variable"]].copy()
143    print(f"Running analysis on {adata_copy.n_vars} highly variable genes")
144    
145    # Initialize results dictionary
146    results_dict = {}
147    
148    # For each method
149    for method in methods:
150        print(f"Processing method: {method}")
151        
152        # Run rank_genes_groups
153        sc.tl.rank_genes_groups(adata_copy, 
154                              groupby=var_cluster,
155                              method=method,
156                              rankby_abs=False,
157                              pts=True,
158                              key_added=method,
159                              n_jobs=n_cores)
160        
161        # Process each cluster
162        method_results = []
163        clusters = sorted(adata_copy.obs[var_cluster].unique())
164        
165        for cluster in clusters:
166            # Get results for this cluster - convert cluster to string to avoid KeyError
167            df = sc.get.rank_genes_groups_df(adata_copy, group=str(cluster), key=method)
168            
169            # Add cluster info and rank
170            df.insert(0, var_cluster, cluster)
171            df['rank'] = range(1, len(df) + 1)
172            
173            # Filter by adjusted p-value if available
174            if 'pvals_adj' in df.columns:
175                df = df[df['pvals_adj'] < 0.05]
176            
177            # Prefix column names with method except names and cluster
178            df.columns = [f'{method}_{col}' if col not in ['names', var_cluster] 
179                         else col for col in df.columns]
180            
181            method_results.append(df)
182        
183        # Combine results for all clusters for this method
184        results_dict[method] = pd.concat(method_results, axis=0)
185    
186    # Merge results from all methods
187    final_df = None
188    for method, df in results_dict.items():
189        if final_df is None:
190            final_df = df
191        else:
192            final_df = final_df.merge(df, on=['names', var_cluster], how='outer')
193    
194    # Keep only one logfoldchange column from methods that have it
195    logfc_methods = [m for m in methods if m != 'logreg']
196    if logfc_methods:
197        logfc_columns = [f'{method}_logfoldchanges' for method in logfc_methods]
198        final_df['logfoldchanges'] = final_df[logfc_columns].mean(axis=1).round(2)
199        final_df = final_df.drop(columns=logfc_columns)
200    
201    # Add rank for logfoldchanges within each cluster
202    if 'logfoldchanges' in final_df.columns:
203        for cluster in final_df[var_cluster].unique():
204            mask = final_df[var_cluster] == cluster
205            final_df.loc[mask, 'logfoldchanges_rank'] = final_df.loc[mask, 'logfoldchanges'].rank(method='min', ascending=False)
206
207    # Keep only one *_pct_nz and *_pct_nz_reference column from methods that have it
208    pct_nz_methods = [m for m in methods if m != 'logreg']
209    if pct_nz_methods:
210        pct_nz_group_columns = [f'{method}_pct_nz_group' for method in pct_nz_methods]
211        pct_nz_reference_columns = [f'{method}_pct_nz_reference' for method in pct_nz_methods]
212        final_df['pct_group'] = (final_df[pct_nz_group_columns].mean(axis=1).round(2))*100
213        final_df['pct_rest'] = (final_df[pct_nz_reference_columns].mean(axis=1).round(2))*100
214        final_df['pct_diff'] = final_df['pct_group'] - final_df['pct_rest']
215        final_df['pct_diff'] = final_df['pct_diff'].round(2)
216        final_df = final_df.drop(columns=pct_nz_group_columns + pct_nz_reference_columns)
217    
218    # Add rank for pct_diff within each cluster
219    if 'pct_diff' in final_df.columns:
220        for cluster in final_df[var_cluster].unique():
221            mask = final_df[var_cluster] == cluster
222            final_df.loc[mask, 'pct_diff_rank'] = final_df.loc[mask, 'pct_diff'].rank(method='min', ascending=False)
223
224    # After merging all methods, calculate cluster-wise median rank
225    rank_columns = [f'{method}_rank' for method in methods]
226    
227    # Group by cluster and calculate median rank within each cluster
228    for cluster in final_df[var_cluster].unique():
229        mask = final_df[var_cluster] == cluster
230        final_df.loc[mask, 'median_rank'] = final_df.loc[mask, rank_columns].median(axis=1)
231        final_df.loc[mask, 'mean_rank'] = final_df.loc[mask, rank_columns].mean(axis=1)
232
233        # Reset rank to be 1-based within each cluster
234        final_df.loc[mask, 'median_rank'] = final_df.loc[mask, 'median_rank'].rank(method='min')
235        final_df.loc[mask, 'mean_rank'] = final_df.loc[mask, 'mean_rank'].rank(method='min')
236
237    # Sort by cluster and median rank
238    final_df = final_df.sort_values([var_cluster, 'median_rank', "mean_rank"])
239    
240    return final_df

236        final_df.loc[mask, 'mean_rank'] = final_df.loc[mask, 'mean_rank'].rank(method='min')
237
238    # Sort by cluster and median rank
239    final_df = final_df.sort_values([var_cluster, 'median_rank', "mean_rank"])
